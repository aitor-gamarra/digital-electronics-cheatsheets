 	\centering\section{\textcolor{MaterialPurple}{Algebra Booleana}}
 	
  	\subsection{\textcolor{GoogleBlue}{Básicos}}
	\begin{itemize}
		\item[\textcolor{MaterialPink}{\textbullet}] \textbf{\textcolor{MaterialPink}{Elemento Neutro:}}
			\subitem \textbullet \hspace{5pt} \textbf{+} $\rightarrow$ 0
			\subitem \textbullet \hspace{5pt} \textbf{$\cdot$} $\rightarrow$ 1
		\item[\textcolor{MaterialPink}{\textbullet}]  \textbf{\textcolor{MaterialPink}{Propiedad Distributiva:}}
			\subitem \textbullet  \hspace{5pt} $a + (b \cdot c)=(a+b) \cdot (a + c)$
			\subitem \textbullet  \hspace{5pt} $a \cdot (b + c)=(a \cdot b) + (a \cdot c)$
		\item[\textcolor{MaterialPink}{\textbullet}]  \textbf{\textcolor{MaterialPink}{Complemento $\rightarrow$} \=a}
			\subitem \textbullet  \hspace{5pt} $a + \bar{a} = 1$
			\subitem \textbullet  \hspace{5pt} $a \cdot \bar{a} = 0$
	\end{itemize}

  	\subsection{\textcolor{GoogleBlue}{Teoremas}}
  
  	\begin{tabular}{l l}
  		 \textbf{\textcolor{MaterialPink}{1.}} $\begin{cases} a + a = a \\	a \cdot a = a \end{cases}$ & \textbf{\textcolor{MaterialPink}{2.}} $\begin{cases}a + \textcolor{MaterialPink}{1 }= \textcolor{MaterialPink}{1 }\\	a \cdot \textcolor{MaterialIndigo}{0 }= \textcolor{MaterialIndigo}{0}\end{cases}$ \\ \\
  		 \textbf{\textcolor{MaterialPink}{3.}} $a = \bar{\bar{a}}$ & \textbf{\textcolor{MaterialPink}{4.}} $\begin{cases}a + ab = a \\	a(a+b) = a\end{cases}$ \\ \\
  		 \textbf{\textcolor{MaterialPink}{5.}} $\begin{cases} a+(b+c) = (a+b)+c\\a(bc) = (ab)c\end{cases}$ & \textbf{\textcolor{MaterialPink}{6.}} $\begin{cases}\overline{(a+b)} = \bar{a}\cdot\bar{b} \\	\overline{a \cdot b} = \bar{a}+\bar{b}\end{cases}$ 	 \\ \\
  	\end{tabular}
  	\begin{flushleft}
  		\hspace{20pt}\bfseries{\textcolor{MaterialIndigo}{Conclusiones}}
  	\end{flushleft}

	\begin{itemize}
		\item[\textcolor{MaterialLightBlue}{\textbullet}] \textcolor{MaterialLightBlue}{\textbf{\scshape{De Morgan (\textcolor{MaterialPink}{6})}}$\rightarrow$} $\begin{cases}\overline{ab} \neq \bar{a}\cdot\bar{b} \\	\overline{a + b} \neq \bar{a}+\bar{b}\end{cases}$ 	 
		\item[\textcolor{MaterialLightBlue}{\textbullet}] \textcolor{MaterialLightBlue}{\textbf{\scshape{Shannon}}$\rightarrow$}$ \overline{f(a,b,\cdots,n,+,\cdot)} =f(\bar{a},\bar{b},\cdots,\bar{n},\cdot,+)$
	\end{itemize}

  	\subsection{\textcolor{GoogleBlue}{Término \scshape{Canónico}}}
  	\begin{itemize}
  		\item[\textcolor{MaterialPink}{\textbullet}]  \textbf{\textcolor{MaterialPink}{\underline{Producto} canónico $\rightarrow$ }}\textcolor{MaterialTeal}{\small\scshape{minterm} = 1}
  			\subitem \begin{equation*}\left.\begin{aligned}\textcolor{MaterialGrey}{000\rightarrow \bar{a}\bar{b}\bar{c}}\\ \textcolor{MaterialGrey}{010 \rightarrow \bar{a}b\bar{c}}\end{aligned}\color{MaterialGrey}\right\} \textcolor{MaterialGrey}{ \Longrightarrow }z=f(a,b,c)=\Sigma(Los \cdot Que Dan 1)\end{equation*}

  			\subitem \begin{equation*}\left.\begin{aligned}\textcolor{MaterialGrey}{ 000 \rightarrow a+b+c}\\\textcolor{MaterialGrey}{ 010 \rightarrow a+\bar{b}+c}\end{aligned}\color{MaterialGrey}\right\} \textcolor{MaterialGrey}{ \Longrightarrow} z=f(a,b,c)=\Pi(Los + Que Dan 0)\end{equation*}
